exports.scheduledPricing = functions.pubsub
  .schedule('every 10 minutes').onRun(async () => {
    const products = await db.collection('products')
      .where('expiresAt', '<=', now + 3600000).get();
    const batch = db.batch();
    products.forEach(doc => {
      const mins = (doc.data().expiresAt - Date.now()) / 60000;
      const discount = mins < 10 ? 70 : mins < 30 ? 62 : 30;
      batch.update(doc.ref, { discount }); // Realtime push
    });
    await batch.commit();
  });
