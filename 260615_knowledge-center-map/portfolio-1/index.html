const fetchProperties = async (sheetId) => {
  const cached = sessionStorage.getItem('prop_cache');
  if (cached && Date.now() - JSON.parse(cached).ts < 5 * 60000)
    return JSON.parse(cached).data; // TTL 5분
  const res = await gapi.client.sheets.spreadsheets.values.get({
    spreadsheetId: sheetId, range: '매물!A2:L'
  });
  const data = res.result.values.map(rowToProperty);
  sessionStorage.setItem('prop_cache', JSON.stringify({ data, ts: Date.now() }));
  return data;
};

매물 설명

성수 IT 지식산업센터 A동 305호는 서울 성동구 성수동 소재 프리미엄 지식산업센터입니다. 전용 66m²의 실용적 평면으로 2~4인 팀에 최적이며, 건물 내 공용 카페테리아·회의실·피트니스가 완비되어 있습니다. 지하철 2호선 성수역 도보 5분으로 교통이 편리하고, 성수 IT 생태계 인접 입지입니다.
